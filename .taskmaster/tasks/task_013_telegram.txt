# Details:
Implement the core message processing pipeline that:

1. Receives incoming messages from webhook
2. Routes to command handler if it's a command
3. Otherwise, processes as a regular message:
   - Updates memory if needed
   - Assembles prompt via Character pipeline
   - Generates response
   - Tracks cost
   - Sends service messages if needed

Example implementation:
```python
class MessageProcessor:
    def __init__(self, character_manager, command_handler, settings_manager, memory_manager, service_message_manager, cost_tracker):
        self.character_manager = character_manager
        self.command_handler = command_handler
        self.settings_manager = settings_manager
        self.memory_manager = memory_manager
        self.service_message_manager = service_message_manager
        self.cost_tracker = cost_tracker
    
    async def process_update(self, persona, update):
        # Extract message and user info
        message = update.get("message", {})
        text = message.get("text", "")
        user_id = message.get("from", {}).get("id")
        chat_id = message.get("chat", {}).get("id")
        
        if not user_id or not chat_id:
            return None
        
        # Get user settings
        settings = self.settings_manager.get_settings(persona, user_id)
        
        # Check if it's a command
        if text.startswith("/"):
            command_parts = text[1:].split(" ", 1)
            command = command_parts[0].lower()
            args = command_parts[1] if len(command_parts) > 1 else ""
            
            response = await self.command_handler.handle_command(persona, user_id, command, args)
            return {"chat_id": chat_id, "response": response}
        
        # Process as regular message
        character = self.character_manager.get_character(persona)
        
        # Generate response
        start_time = time.time()
        response_data = await character.respond(user_id, text, settings)
        response_text = response_data.get("response", "")
        processing_time = time.time() - start_time
        
        # Track cost
        model = settings.get("model", "gpt-4o")
        input_tokens = response_data.get("input_tokens", 0)
        output_tokens = response_data.get("output_tokens", 0)
        self.cost_tracker.record_usage(user_id, model, input_tokens, output_tokens)
        
        # Check for memory updates
        memory_updates = response_data.get("memory_updates", [])
        service_messages = []
        
        for memory in memory_updates:
            self.memory_manager.add_memory(
                persona, user_id, memory.get("content"), memory.get("summary"), memory.get("weight", 1.0)
            )
            
            # Generate service message if enabled
            if settings.get("service_messages", True):
                service_msg = self.service_message_manager.notify_memory_added(
                    persona, user_id, memory.get("summary"), memory.get("weight", 1.0)
                )
                if service_msg:
                    service_messages.append(service_msg)
        
        # Check for relationship updates
        relationship_updates = response_data.get("relationship_updates", [])
        for update in relationship_updates:
            result = self.memory_manager.update_relationship(
                persona, user_id, update.get("axis"), update.get("delta", 0)
            )
            
            # Generate service message if enabled
            if settings.get("service_messages", True) and abs(update.get("delta", 0)) >= 0.05:
                service_msg = self.service_message_manager.notify_relationship_change(
                    persona, user_id, result.get("axis"), result.get("old"), result.get("new")
                )
                if service_msg:
                    service_messages.append(service_msg)
        
        # Check if we should send cost summary
        if settings.get("cost_summaries", True) and self.cost_tracker.should_send_daily_summary(user_id, settings):
            cost_msg = self.cost_tracker.format_cost_message(user_id, settings)
            service_messages.append(cost_msg)
            self.cost_tracker.mark_daily_summary_sent(user_id)
        
        # Prepare response with service messages
        if service_messages and settings.get("service_messages", True):
            response_text += "\n\n" + "\n".join(service_messages)
        
        return {
            "chat_id": chat_id,
            "response": response_text,
            "processing_time": processing_time
        }
```
