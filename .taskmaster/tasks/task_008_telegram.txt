# Description: Create a system to track and report token usage and associated costs.
# Details:
Implement a CostTracker class that:

1. Records per-call usage (tokens and cost)
2. Calculates session and monthly totals
3. Generates cost reports for /cost command
4. Provides daily summaries if enabled

Follow the structure in docs/cost_tracker.md as mentioned in the PRD.

Example implementation:
```python
class CostTracker:
    def __init__(self, data_dir="backend/data/telegram"):
        self.data_dir = data_dir
        os.makedirs(data_dir, exist_ok=True)
        
        # Cost per 1K tokens by model (approximate)
        self.cost_per_1k = {
            "gpt-4o": {"input": 0.005, "output": 0.015},
            "gpt-4.1-turbo": {"input": 0.01, "output": 0.03},
            "sonnet-3.7": {"input": 0.003, "output": 0.006}
        }
    
    def _get_cost_path(self, user_id):
        return os.path.join(self.data_dir, "costs", f"{user_id}.jsonl")
    
    def record_usage(self, user_id, model, input_tokens, output_tokens):
        path = self._get_cost_path(user_id)
        os.makedirs(os.path.dirname(path), exist_ok=True)
        
        # Calculate cost
        input_cost = (input_tokens / 1000) * self.cost_per_1k.get(model, {}).get("input", 0.01)
        output_cost = (output_tokens / 1000) * self.cost_per_1k.get(model, {}).get("output", 0.03)
        total_cost = input_cost + output_cost
        
        record = {
            "timestamp": datetime.now().isoformat(),
            "model": model,
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
            "total_tokens": input_tokens + output_tokens,
            "input_cost": input_cost,
            "output_cost": output_cost,
            "total_cost": total_cost
        }
        
        with open(path, "a") as f:
            f.write(json.dumps(record) + "\n")
        
        return record
    
    def get_session_cost(self, user_id, hours=24):
        path = self._get_cost_path(user_id)
        if not os.path.exists(path):
            return {"total_tokens": 0, "total_cost": 0.0}
        
        cutoff = datetime.now() - timedelta(hours=hours)
        session_tokens = 0
        session_cost = 0.0
        
        with open(path, "r") as f:
            for line in f:
                record = json.loads(line)
                record_time = datetime.fromisoformat(record["timestamp"])
                
                if record_time >= cutoff:
                    session_tokens += record["total_tokens"]
                    session_cost += record["total_cost"]
        
        return {
            "total_tokens": session_tokens,
            "total_cost": session_cost
        }
    
    def get_monthly_cost(self, user_id):
        path = self._get_cost_path(user_id)
        if not os.path.exists(path):
            return {"total_tokens": 0, "total_cost": 0.0}
        
        now = datetime.now()
        month_start = datetime(now.year, now.month, 1)
        
        monthly_tokens = 0
        monthly_cost = 0.0
        
        with open(path, "r") as f:
            for line in f:
                record = json.loads(line)
                record_time = datetime.fromisoformat(record["timestamp"])
                
                if record_time >= month_start:
                    monthly_tokens += record["total_tokens"]
                    monthly_cost += record["total_cost"]
        
        return {
            "total_tokens": monthly_tokens,
            "total_cost": monthly_cost
        }
    
    def format_cost_message(self, user_id, settings):
        session = self.get_session_cost(user_id)
        monthly = self.get_monthly_cost(user_id)
        
        # Format with thousands separator
        session_tokens = f"{session['total_tokens']:,}".replace(",", " ")
        session_cost = f"${session['total_cost']:.3f}"
        monthly_cost = f"${monthly['total_cost']:.2f}"
        
        if settings.get("language", "en") == "ru":
            return f"💰 Сессия: {session_tokens} ток. ≈ {session_cost} | Месяц: {monthly_cost}"
        else:
            return f"💰 Session: {session_tokens} tokens ≈ {session_cost} | Month: {monthly_cost}"
    
    def should_send_daily_summary(self, user_id, settings):
        if not settings.get("cost_summaries", True):
            return False
        
        # Check if we've sent a summary today
        summary_path = os.path.join(self.data_dir, "cost_summaries", f"{user_id}.json")
        
        if os.path.exists(summary_path):
            with open(summary_path, "r") as f:
                data = json.load(f)
                last_summary = datetime.fromisoformat(data.get("last_summary", "2000-01-01"))
                today = datetime.now().replace(hour=0, minute=0, second=0, microsecond=0)
                
                if last_summary >= today:
                    return False
        
        return True
    
    def mark_daily_summary_sent(self, user_id):
        summary_path = os.path.join(self.data_dir, "cost_summaries", f"{user_id}.json")
        os.makedirs(os.path.dirname(summary_path), exist_ok=True)
        
        with open(summary_path, "w") as f:
            json.dump({"last_summary": datetime.now().isoformat()}, f)
```
